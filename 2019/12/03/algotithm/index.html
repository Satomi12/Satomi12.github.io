kmp 算法


public int kmp(String s, String p) {
    int[] next = getNext(p);
    int i = 0, j = 0;
    while (i < s.length() && j < p.length()) {
        if (j == -1 || s.charAt(i) == p.charAt(j)) {
            i++;
            j++;
        } else {
            j = next[j];
        }
    }
    return j == p.length() ? i - j : -1;
}

public int[] getNext(String p) {
    if (p.length() == 0) return new int[0];
    int[] next = new int[p.length()];
    next[0] = -1;
    int i = 0, j = -1;
    while (i < p.length() - 1) {
        if (j == -1 || p.charAt(i) == p.charAt(j)) {
            i++;
            j++;
            next[i] = j;
        } else {
            j = next[j];
        }
    }
    return next;
}

字典树


class TrieNode {
    String word;
    TrieNode[] next = new TrieNode[26];
}

public TrieNode buildTrie(String[] words) {
    TrieNode root = new TrieNode();
    for (String word : words) {
        TrieNode p = root;
        for(char c : word.toCharArray()) {
            if (p.next[c-'a'] == null) {
                p.next[c-'a'] = new TrieNode();
            }
            p = p.next[c-'a'];
        }
        p.word = word;
    }
    return root;
}

Manacher 算法:


public String manacher(String s) {
    StringBuilder sb = new StringBuilder();
    sb.append("$#");
    for (char c : s.toCharArray()) {
        sb.append(c + "#");
    }
    sb.append('#');
    int[] p = new int[sb.length()];
    int id = 0, mx = 0;
    int maxLen = 0, start = -1;
    for (int i = 1; i < sb.length() - 1; i++) {
        if (mx > i) p[i] = Math.min(p[2 * id - i], mx - i);
        else p[i] = 1;
        while (sb.charAt(i + p[i]) == sb.charAt(i - p[i])) {
            p[i]++;
        }
        if (i + p[i] > mx) {
            id = i;
            mx = i + p[i];
        }
        if (p[i] - 1 > maxLen) {
            maxLen = p[i] - 1;
            start = (i - p[i]) / 2;
        }
    }
    return s.substring(start, start + maxLen);
}

public int add(int a, int b) {

    }
    return a;
}

public int multiply(int a, int b) {
    int one = a < 0 ? add(~a, 1) : a;
    int two = b < 0 ? add(~b, 1) : b;
    int product = 0;
    while (two > 0) {
        if ((two & 1) > 0) {
            product = add(product, one);
        }
        one <<= 1; two>>= 1;
    }
    return (a ^ b) < 0 ? add(~product, 1) : product;
}

位除法


public int add(int a, int b) {
    while (b != 0) {
        int carry = a & b;
        a ^= b;
        b = carry << 1;
    }
    return a;
}

public int sub(int a, int b) {
    while (b != 0) {
        int borrow = ~a & b;
        a ^= b;
        b = borrow << 1;
    }
    return a;
}

public int divide(int a, int b) {
    int dividend = a < 0 ? add(~a, 1) : a;
    int divisor = b < 0 ? add(~b, 1) : b;
    int quotient = 0, remainder = 0;
    for (int i = 31; i >= 0; i--) {
        if ((dividend >> i) >= divisor) {
            quotient = add(quotient, 1 << i);
            dividend = sub(dividend, divisor << i);
        }
    }
    // 这里返回的结果是取模
    remainder = b < 0 ? add(~dividend, 1) : dividend;
    return (a ^ b) < 0 ? add(~quotient, 1) : quotient;
}

数论

欧拉筛法


public static void eulerSeive(int n) {
    boolean[] isPrime = new boolean[n + 1];
    int[] prime = new int[n];
    int cnt = 0;
    isPrime[0] = isPrime[1] = true;
    for (int i = 2; i <= n; i++) {
        if (!isPrime[i]) {
            prime[cnt++] = i;
        }
        for (int j = 0; j < cnt; j++) {
            if (i * prime[j] > n) break;
            isPrime[i * prime[j]] = true;
            if (i % prime[j] == 0) break;
        }
    }
}

先序

Morris Traverse 版本


public List preorderTraversal(TreeNode root) {
    List res = new ArrayList<>();
    while (root != null) {
        if (root.left == null) {
            res.add(root.val);
            root = root.right;
        } else {
            TreeNode pre = root.left;
            while (pre.right != null && pre.right != root) {
                pre = pre.right;
            }
            if (pre.right == null) {
                res.add(root.val);
                pre.right = root;
                root = root.left;
            } else {
                pre.right = null;
                root = root.right;
            }
        }
    }
    return res;
}
